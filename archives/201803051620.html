    Python内置变量类型

    浮点

    float_number = 2.3

    复数

    complex_number = 1 + 2j

    整型

    integer_number = 1

    列表

    sample_list = [1,23,456,'abc']

    嵌套

    sample_nest = [(2,4,6),{5:7,9:11,'key':[2,5]},6]

    字典

    sample_dict = {"key":value,5:100}

    元组

    sample_tuple = (1,2,'ab')
